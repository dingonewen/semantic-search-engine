<script>
  function shortName(id) {
    return id.split('/').pop();
  }

  async function fetchSnippet(id) {
    try {
      const r = await fetch('/static/' + id);
      if (!r.ok) return '';
      const t = await r.text();
      return t.trim().slice(0, 180);
    } catch { return ''; }
  }

  function esc(s) {
    return s.replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;');
  }

  // RRF scores are ~0–0.033; map to 0–100% bar width
  function scorePercent(s) {
    return Math.min(Math.max(s * 3000, 6), 100).toFixed(0);
  }

  async function renderCards(results) {
    const el = document.getElementById('search-results');
    if (!results || results.length === 0) {
      el.innerHTML = '<p class="error-msg">No results found.</p>';
      return;
    }
    el.innerHTML = '<div class="cards" id="cards-list"></div>';
    const list = document.getElementById('cards-list');

    for (const r of results) {
      const snippet = await fetchSnippet(r.id);
      const pct = scorePercent(r.score);
      const card = document.createElement('div');
      card.className = 'card';
      card.innerHTML =
        '<div class="card-filename">' + esc(shortName(r.id)) + '</div>' +
        '<div class="score-row">' +
          '<div class="score-track"><div class="score-fill" style="width:' + pct + '%"></div></div>' +
          '<span class="score-val">' + r.score.toFixed(4) + '</span>' +
        '</div>' +
        (snippet ? '<div class="card-snippet">' + esc(snippet) + '</div>' : '');
      list.appendChild(card);
    }
  }

  document.getElementById('search-form').addEventListener('submit', async (e) => {
    e.preventDefault();
    const q = document.getElementById('search-input').value.trim();
    if (!q) return;
    const el = document.getElementById('search-results');
    el.innerHTML = '<div class="loading"><div class="spinner"></div>Searching…</div>';
    document.getElementById('ask-result').innerHTML = '<p class="placeholder">Ask a question and the AI will synthesize an answer from your documents.</p>';
    try {
      const res = await fetch('/api/search?terms=' + encodeURIComponent(q));
      const data = await res.json();
      await renderCards(data.hybrid);
    } catch (err) {
      el.innerHTML = '<p class="error-msg">Search failed: ' + esc(err.message) + '</p>';
    }
  });

  document.getElementById('ask-form').addEventListener('submit', async (e) => {
    e.preventDefault();
    const q = document.getElementById('ask-input').value.trim();
    if (!q) return;
    const el = document.getElementById('ask-result');
    el.innerHTML = '<div class="loading"><div class="spinner"></div>Thinking…</div>';
    try {
      const res = await fetch('/api/ask?q=' + encodeURIComponent(q));
      const data = await res.json();
      if (data.error) {
        el.innerHTML = '<p class="error-msg">' + esc(data.error) + '</p>';
        return;
      }
      let html = '<div class="answer-box"><div class="answer-text">' + esc(data.answer) + '</div>';
      if (data.sources && data.sources.length) {
        html += '<div class="sources-section"><span class="sources-heading">Sources</span>';
        for (const s of data.sources) {
          html += '<a class="source-link" href="/static/' + s + '">' + esc(shortName(s)) + '</a>';
        }
        html += '</div>';
      }
      html += '</div>';
      el.innerHTML = html;
    } catch (err) {
      el.innerHTML = '<p class="error-msg">Ask failed: ' + esc(err.message) + '</p>';
    }
  });
</script>
